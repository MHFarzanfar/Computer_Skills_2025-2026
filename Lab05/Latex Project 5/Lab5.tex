	\end{tabular}
	
	\subsubsection*{3. Testing table structure}
	\textbf{Correct table:}
	\begin{tabular}{ccc}
		A & B & C \\
		X & Y & Z \\
	\end{tabular}
	
	\textbf{Too few items (error):}
	\begin{verbatim}
		% \begin{tabular}{ccc}
			%   A & B & C \\
			%   X & Y     % Missing third element
			% \end{tabular}
	\end{verbatim}
	
	\textbf{Too many items (error):}
	\begin{verbatim}
		% \begin{tabular}{cc}
			%   1 & 2 & 3 & 4  % Too many elements
			% \end{tabular}
	\end{verbatim}
	
	\subsubsection*{4. Multicolumn spanning}
